# Theme & Appearance Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Acorn에 theme(빌트인 7개 CSS + .css drop-in), 단일 배경 이미지(fit/opacity/blur + App/Terminal 영역 토글), 큐레이션 모노스페이스 폰트 3슬롯 dropdown을 추가한다.

**Architecture:** Tailwind v4 `@theme` 블록은 변수 *이름*과 컴파일 타임 기본값을 유지(유틸리티 클래스 생성용). 런타임 theme 전환은 `<style id="acorn-theme">`에 `:root[data-acorn-theme="<id>"]` selector로 동일 변수를 override해서 적용. 배경 이미지는 `$APPLOCALDATA/backgrounds/` 아래 단일 파일로 보관, CSS 변수 4개로 fit/opacity/blur 제어. 폰트는 `appearance.fontSlots`가 truth이고 기존 `terminal.fontFamily`는 거기서 derive.

**Tech Stack:** React 19 + Tailwind v4, Zustand, Tauri 2 (plugin-fs, plugin-dialog, plugin-opener, api/path/convertFileSrc), Vitest (jsdom).

**Spec:** `docs/superpowers/specs/2026-05-13-theme-design.md`

```ts
export const CURATED_MONOSPACE_FONTS = [
  "JetBrains Mono",
  "Fira Code",
  "Cascadia Code",
  "SF Mono",
  "Menlo",
  "Monaco",
  "Consolas",
  "IBM Plex Mono",
  "Source Code Pro",
  "Hack",
] as const;

export type CuratedMonospaceFont = (typeof CURATED_MONOSPACE_FONTS)[number];

export type FontSlots = [string | null, string | null, string | null];

const GENERIC_FALLBACKS = new Set([
  "monospace",
  "sans-serif",
  "serif",
  "system-ui",
  "ui-monospace",
  "ui-sans-serif",
  "ui-serif",
]);

function needsQuoting(name: string): boolean {
  return /\s/.test(name);
}

export function fontStackFromSlots(
  slots: Array<string | null | undefined>,
  fallback: string,
): string {
  const cleaned = slots
    .filter((s): s is string => !!s && s.trim().length > 0)
    .map((s) => s.trim());
  const parts = cleaned.map((name) =>
    needsQuoting(name) ? `"${name}"` : name,
  );
  parts.push(fallback);
  return parts.join(", ");
}

export function fontSlotsFromStack(stack: string): string[] {
  return stack
    .split(",")
    .map((token) => token.trim().replace(/^"(.*)"$/, "$1"))
    .filter((name) => name.length > 0 && !GENERIC_FALLBACKS.has(name))
    .slice(0, 3);
}
```

`src/lib/themes.ts`:

```ts
import { open as openPath } from "@tauri-apps/plugin-opener";
import { appLocalDataDir, join } from "@tauri-apps/api/path";
import { readDir, readTextFile, mkdir, exists } from "@tauri-apps/plugin-fs";

import acornDarkCss from "../assets/themes/acorn-dark.css?raw";
import oneDarkProCss from "../assets/themes/one-dark-pro.css?raw";
import tokyoNightCss from "../assets/themes/tokyo-night.css?raw";
import acornLightCss from "../assets/themes/acorn-light.css?raw";
import githubLightCss from "../assets/themes/github-light.css?raw";
import solarizedLightCss from "../assets/themes/solarized-light.css?raw";
import catppuccinLatteCss from "../assets/themes/catppuccin-latte.css?raw";

export type ThemeMode = "dark" | "light";

export interface AcornTheme {
  id: string;
  label: string;
  mode: ThemeMode;
  /** Raw CSS text. Must contain a `:root[data-acorn-theme="<id>"]` block. */
  css: string;
  /** Provenance — used for UI badges and to allow user overrides. */
  source: "builtin" | "user";
}

export const THEME_CSS_VARS = [
  "--color-bg",
  "--color-bg-elevated",
  "--color-bg-sidebar",
  "--color-fg",
  "--color-fg-muted",
  "--color-border",
  "--color-accent",
  "--color-accent-hover",
  "--color-danger",
  "--color-warning",
  "--color-terminal-bg",
  "--color-terminal-fg",
] as const;

export const BUILT_IN_THEMES: ReadonlyArray<AcornTheme> = [
  { id: "acorn-dark", label: "Acorn Dark", mode: "dark", css: acornDarkCss, source: "builtin" },
  { id: "one-dark-pro", label: "One Dark Pro", mode: "dark", css: oneDarkProCss, source: "builtin" },
  { id: "tokyo-night", label: "Tokyo Night", mode: "dark", css: tokyoNightCss, source: "builtin" },
  { id: "acorn-light", label: "Acorn Light", mode: "light", css: acornLightCss, source: "builtin" },
  { id: "github-light", label: "GitHub Light", mode: "light", css: githubLightCss, source: "builtin" },
  { id: "solarized-light", label: "Solarized Light", mode: "light", css: solarizedLightCss, source: "builtin" },
  { id: "catppuccin-latte", label: "Catppuccin Latte", mode: "light", css: catppuccinLatteCss, source: "builtin" },
];

export type ValidateResult =
  | { ok: true }
  | { ok: false; missing: string[] };

export function validateThemeCss(css: string): ValidateResult {
  const missing: string[] = [];
  for (const v of THEME_CSS_VARS) {
    const re = new RegExp(`${v}\\s*:\\s*[^;\\n]+`);
    if (!re.test(css)) missing.push(v);
  }
  return missing.length === 0 ? { ok: true } : { ok: false, missing };
}

const STYLE_ELEMENT_ID = "acorn-theme";

export function applyTheme(id: string, css: string): void {
  let styleEl = document.getElementById(STYLE_ELEMENT_ID) as
    | HTMLStyleElement
    | null;
  if (!styleEl) {
    styleEl = document.createElement("style");
    styleEl.id = STYLE_ELEMENT_ID;
    document.head.appendChild(styleEl);
  }
  styleEl.textContent = css;
  document.documentElement.setAttribute("data-acorn-theme", id);
}

const USER_THEMES_DIR = "themes";

async function ensureThemesDir(): Promise<string> {
  const root = await appLocalDataDir();
  const dir = await join(root, USER_THEMES_DIR);
  if (!(await exists(dir))) {
    await mkdir(dir, { recursive: true });
  }
  return dir;
}

/**
 * Scan `$APPLOCALDATA/themes/*.css`, validate, and return AcornTheme entries.
 * Filename (without `.css`) becomes the id; label is derived by replacing
 * dashes with spaces and title-casing.
 */
export async function loadUserThemes(): Promise<AcornTheme[]> {
  try {
    const dir = await ensureThemesDir();
    const entries = await readDir(dir);
    const out: AcornTheme[] = [];
    for (const entry of entries) {
      if (!entry.isFile || !entry.name?.endsWith(".css")) continue;
      const id = entry.name.replace(/\.css$/, "");
      const path = await join(dir, entry.name);
      const css = await readTextFile(path);
      const result = validateThemeCss(css);
      if (!result.ok) {
        console.warn(
          `[acorn] skipping theme ${entry.name}: missing ${result.missing.join(", ")}`,
        );
        continue;
      }
      out.push({
        id,
        label: humanize(id),
        mode: css.includes("/* @mode light */") ? "light" : "dark",
        css,
        source: "user",
      });
    }
    return out;
  } catch (err) {
    console.warn("[acorn] failed to load user themes", err);
    return [];
  }
}

function humanize(id: string): string {
  return id
    .split("-")
    .map((w) => (w.length ? w[0].toUpperCase() + w.slice(1) : w))
    .join(" ");
}

export async function revealThemesFolder(): Promise<void> {
  const dir = await ensureThemesDir();
  await openPath(dir);
}

/**
 * Built-in + user themes merged. User entries with the same id as a built-in
 * override the built-in. Stable order: built-ins first (in declaration order),
 * then user themes sorted by id.
 */
export function mergeThemes(
  builtin: ReadonlyArray<AcornTheme>,
  user: ReadonlyArray<AcornTheme>,
): AcornTheme[] {
  const userById = new Map(user.map((t) => [t.id, t]));
  const merged: AcornTheme[] = builtin.map((t) => userById.get(t.id) ?? t);
  const extras = user
    .filter((t) => !builtin.some((b) => b.id === t.id))
    .sort((a, b) => a.id.localeCompare(b.id));
  return [...merged, ...extras];
}
```

import { convertFileSrc } from "@tauri-apps/api/core";
import { appLocalDataDir, join } from "@tauri-apps/api/path";
import {
  exists,
  mkdir,
  readDir,
  remove,
  writeFile,
} from "@tauri-apps/plugin-fs";

export type BackgroundFit = "cover" | "contain" | "tile";

export interface BackgroundState {
  relativePath: string | null;
  fileName: string | null;
  fit: BackgroundFit;
  opacity: number; // 0..1
  blur: number;    // 0..24
  applyToApp: boolean;
  applyToTerminal: boolean;
}

export const BG_DIR = "backgrounds";

async function ensureBackgroundsDir(): Promise<string> {
  const root = await appLocalDataDir();
  const dir = await join(root, BG_DIR);
  if (!(await exists(dir))) {
    await mkdir(dir, { recursive: true });
  }
  return dir;
}

function shortHash(bytes: Uint8Array): string {
  // FNV-1a 32-bit → 8 hex chars. Cryptographic strength unnecessary; this is
  // a filename disambiguator, not a security boundary.
  let h = 0x811c9dc5;
  for (let i = 0; i < bytes.length; i++) {
    h ^= bytes[i];
    h = Math.imul(h, 0x01000193);
  }
  return (h >>> 0).toString(16).padStart(8, "0");
}

function extOf(name: string): string {
  const m = name.match(/\.([a-zA-Z0-9]+)$/);
  return m ? `.${m[1].toLowerCase()}` : "";
}

/**
 * Copy a chosen image into `$APPLOCALDATA/backgrounds/<hash><ext>`, removing
 * any previously-stored backgrounds first (single-image policy).
 */
export async function importBackgroundImage(
  originalName: string,
  bytes: Uint8Array,
): Promise<{ relativePath: string; fileName: string }> {
  const dir = await ensureBackgroundsDir();

  // Clean previous files (single-image policy)
  try {
    const entries = await readDir(dir);
    for (const entry of entries) {
      if (entry.isFile && entry.name) {
        const p = await join(dir, entry.name);
        await remove(p).catch(() => {});
      }
    }
  } catch {
    // dir was freshly created; nothing to clean
  }

  const fileName = `${shortHash(bytes)}${extOf(originalName)}`;
  const absolute = await join(dir, fileName);
  await writeFile(absolute, bytes);
  return {
    relativePath: `${BG_DIR}/${fileName}`,
    fileName: originalName,
  };
}

export async function removeBackgroundImage(
  relativePath: string,
): Promise<void> {
  const root = await appLocalDataDir();
  const absolute = await join(root, relativePath);
  await remove(absolute).catch(() => {});
}

export function backgroundCssVarsForState(
  s: BackgroundState,
): Record<string, string> {
  if (!s.relativePath) {
    return {
      "--bg-image-url": "none",
      "--bg-fit-size": s.fit === "tile" ? "auto" : s.fit,
      "--bg-fit-repeat": s.fit === "tile" ? "repeat" : "no-repeat",
      "--bg-opacity": String(s.opacity),
      "--bg-blur": `${s.blur}px`,
    };
  }
  // relativePath is portable; resolve to absolute lazily at apply-time.
  // For unit tests we accept "asset://" via mock; runtime convertFileSrc
  // accepts either filesystem path or file:// URL.
  return {
    "--bg-image-url": `url("PLACEHOLDER")`, // replaced below in apply
    "--bg-fit-size": s.fit === "tile" ? "auto" : s.fit,
    "--bg-fit-repeat": s.fit === "tile" ? "repeat" : "no-repeat",
    "--bg-opacity": String(s.opacity),
    "--bg-blur": `${s.blur}px`,
  };
}

async function resolveImageUrl(relativePath: string): Promise<string> {
  const root = await appLocalDataDir();
  const absolute = await join(root, relativePath);
  return convertFileSrc(absolute);
}

export async function applyBackgroundVars(s: BackgroundState): Promise<void> {
  const vars = backgroundCssVarsForState(s);
  if (s.relativePath) {
    const url = await resolveImageUrl(s.relativePath);
    vars["--bg-image-url"] = `url("${url}")`;
  }
  for (const [k, v] of Object.entries(vars)) {
    document.documentElement.style.setProperty(k, v);
  }
  document.documentElement.setAttribute(
    "data-bg-app",
    s.relativePath && s.applyToApp ? "on" : "off",
  );
  document.documentElement.setAttribute(
    "data-bg-terminal",
    s.relativePath && s.applyToTerminal ? "on" : "off",
  );
}

export function clearBackgroundVars(): void {
  for (const k of [
    "--bg-image-url",
    "--bg-fit-size",
    "--bg-fit-repeat",
    "--bg-opacity",
    "--bg-blur",
  ]) {
    document.documentElement.style.removeProperty(k);
  }
  document.documentElement.removeAttribute("data-bg-app");
  document.documentElement.removeAttribute("data-bg-terminal");
}

const VALID_BG_FIT = new Set<BackgroundFit>(["cover", "contain", "tile"]);
const CURATED_SET = new Set<string>(CURATED_MONOSPACE_FONTS);
const VALID_THEME_IDS = new Set<string>(BUILT_IN_THEMES.map((t) => t.id));

function normalizeBgFit(v: unknown, fallback: BackgroundFit): BackgroundFit {
  return typeof v === "string" && VALID_BG_FIT.has(v as BackgroundFit)
    ? (v as BackgroundFit)
    : fallback;
}

function clamp01(v: unknown, fallback: number): number {
  if (typeof v !== "number" || !Number.isFinite(v)) return fallback;
  return Math.max(0, Math.min(1, v));
}

function clampBlur(v: unknown, fallback: number): number {
  if (typeof v !== "number" || !Number.isFinite(v)) return fallback;
  return Math.max(0, Math.min(24, v));
}

function normalizeThemeId(v: unknown, fallback: string): string {
  // Built-in IDs are validated against the registry; unknown ids reset.
  // (User-loaded themes are validated later in the boot path; if their id
  // matches a stored value, they'll re-take effect on next save.)
  return typeof v === "string" && VALID_THEME_IDS.has(v) ? v : fallback;
}

function normalizeFontSlots(
  v: unknown,
  fallback: AcornSettings["appearance"]["fontSlots"],
): AcornSettings["appearance"]["fontSlots"] {
  if (!Array.isArray(v)) return fallback;
  const cleaned: Array<string | null> = [0, 1, 2].map((i) => {
    const raw = v[i];
    if (typeof raw !== "string") return null;
    return CURATED_SET.has(raw) ? raw : null;
  });
  if (!cleaned[0]) return fallback;
  return [cleaned[0]!, cleaned[1], cleaned[2]];
}

```tsx
function FontSection({
  slots,
  onChange,
}: {
  slots: [string, string | null, string | null];
  onChange: (slots: [string, string | null, string | null]) => void;
}) {
  const setSlot = (idx: 0 | 1 | 2, value: string) => {
    const next: [string, string | null, string | null] = [...slots] as [
      string,
      string | null,
      string | null,
    ];
    const v = value === "" ? null : value;
    if (idx === 0) {
      if (!v) return; // primary required
      next[0] = v;
    } else {
      next[idx] = v;
    }
    onChange(next);
  };

  const renderSlot = (idx: 0 | 1 | 2, label: string, required: boolean) => (
    <Field key={idx} label={label}>
      <Select
        value={slots[idx] ?? ""}
        onChange={(e) => setSlot(idx, e.target.value)}
        className="min-w-[14rem]"
      >
        {!required ? <option value="">— (skip)</option> : null}
        {CURATED_MONOSPACE_FONTS.map((f) => (
          <option key={f} value={f}>
            {f}
          </option>
        ))}
      </Select>
    </Field>
  );

  return (
    <Field
      label="Terminal font"
      hint="Primary 부터 순서대로 fallback. 마지막에는 자동으로 generic monospace 가 붙음."
    >
      <div className="space-y-2">
        {renderSlot(0, "Primary", true)}
        {renderSlot(1, "Secondary", false)}
        {renderSlot(2, "Tertiary", false)}
      </div>
    </Field>
  );
}
```
